class Solution {
public:
    int search(vector<int>& nums, int target) {
        int s=0;
        int pos=0;
        for(int i=0;i<=nums.size()-1;i++){
            if(nums[i]==target){
             s=1;
             pos=i;
             break;
             }
        }
        if(s==0) return-1;
        else return pos;
    }
};
